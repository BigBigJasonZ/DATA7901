import folium
from folium.plugins import MarkerCluster
import pandas as pd

data = pd.read_excel('all.xlsx')
m = folium.Map(location=[26, -90], zoom_start=2)
marker_cluster = MarkerCluster().add_to(m)

for index, row in data.iterrows():
    longitude = row['longitude']
    latitude = row['latitude']
    launch_site = row['launch_site']
    outcome_result = row['outcome_result']
    date_utc = row['date_utc']
    outcome = row['outcome']

    if outcome_result == 1:
        icon_c = 'green'
        outcome_text = 'SUCCESS'
    else:
        icon_c = 'red'
        outcome_text = 'FALSE'

    marker = folium.Marker(
        location=[latitude, longitude],
        tooltip=launch_site,
        icon=folium.Icon(icon_c)
    )

    circle = folium.Circle(
        location=[latitude, longitude],
        radius=500,
        color='orange',
        fill=True,
        fill_color='yellow',
        fill_opacity=0.01
    )
    circle.add_to(m)

    mouse_popup_text = f"Date_utc:{date_utc}<br>Launch Site: {launch_site}<br>Outcome:{outcome}<br>Outcome result:{outcome_text}"
    popup = folium.Popup(mouse_popup_text, max_width=300)
    popup.add_to(marker)
    marker.add_to(marker_cluster)

m.save('folium_map.html')

import dash
from dash import dcc, html
import pandas as pd

df = pd.read_excel("all.xlsx")
app = dash.Dash(__name__)

# define grid
app.layout = html.Div([
    dcc.Dropdown(
        id='launch-site-dropdown',
        options=[
            {'label': site, 'value': site}
            for site in df['launch_site'].unique()
        ],
        value=df['launch_site'].unique()[0],  
        multi=False  
    ),
    dcc.Graph(
        id='success-rate-pie-chart'
    )
])

# Create Callback Function
@app.callback(
    dash.dependencies.Output('success-rate-pie-chart', 'figure'),
    [dash.dependencies.Input('launch-site-dropdown', 'value')]
)
def update_pie_chart(selected_site):
    filtered_df = df[df['launch_site'] == selected_site]
    success_rate = filtered_df['outcome_result'].mean()
   
    fig = px.pie(
        names=['success', 'fail'],
        values=[success_rate, 1 - success_rate],
        title=f'{selected_site} success rate',
        hole=0.5
    )
   
    return fig

if __name__ == '__main__':
    app.run_server(debug=True)

import dash
from dash import dcc, html
from dash.dependencies import Input, Output
import plotly.express as px
import pandas as pd


df = pd.read_excel('all.xlsx')
app = dash.Dash(__name__)

app.layout = html.Div([
    dcc.RangeSlider(
        id='payload-mass-slider',
        marks={i: str(i) for i in range(0, 20000, 1000)},
        min=0,
        max=20000,
        step=1000,
        value=[0, 20000],
    ),
    dcc.Graph(id='payload-mass-comparison'),

])

@app.callback(
    Output('payload-mass-comparison', 'figure'),
    Input('payload-mass-slider', 'value')
)
def update_payload_mass_comparison(selected_range):
    filtered_df = df[(df['payload_mass'] >= selected_range[0]) & (df['payload_mass'] <= selected_range[1])]
    fig = px.scatter(filtered_df, x='launch_site', y='payload_mass')
    
    return fig


if __name__ == '__main__':
    app.run_server(debug=True)
